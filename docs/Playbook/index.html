Class Playbook 

open class Playbook

The central scenario manager.

You can use the singleton default or use it with instantiate arbitrarily.

struct Content: View {
    var body: some View {
        Text("Playbook")
    }
}

Playbook.default.addScenarios(of: "Components") {
    Scenario("Content", layout: .compressed) {
        Content()
    }
}
